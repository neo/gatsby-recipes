Add npm scripts

<NPMScript name="develop" command="GATSBY_HOT_LOADER=fast-refresh gatsby develop" />
<NPMScript name="build" command="gatsby build" />
<NPMScript name="serve" command="gatsby serve" />
<NPMScript name="clean" command="gatsby clean" />
<NPMScript name="start" command="npm run develop" />

---

Install minimum Gatsby, React and React DOM

<NPMPackage name="gatsby" />
<NPMPackage name="react" />
<NPMPackage name="react-dom" />

---

Add TypeScript

<NPMPackage name="typescript" />
<NPMPackage name="gatsby-plugin-ts" />
<GatsbyPlugin
  name="gatsby-plugin-ts"
  options={{
    codegen: false,
    forkTsCheckerPlugin: {
      eslint: true
    }
  }}
/>

---

Add ESLint

<NPMPackage name="eslint" />
<NPMPackage name="eslint-config-typescript-react-prettier" />
<File path=".eslintrc" content={`{
  "extends": ["typescript-react-prettier"]
}
`} />

---

Add SCSS

<NPMPackage name="node-sass" />
<NPMPackage name="gatsby-plugin-sass" />
<GatsbyPlugin
  name="gatsby-plugin-sass"
  options={{
    cssLoaderOptions: {
      localIdentName: "[emoji]",
    },
  }}
/>

That's it! Remember to initialize your `tsconfig.json` file with:

`npx tsc --init --noEmit --target esnext --jsx react`
